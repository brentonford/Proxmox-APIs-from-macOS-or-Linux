VM_NAME="{newOsNode.fqn}"
PVE_HOST="{proxmoxNode.ip}"
ISO="kickstart_{newOsNode.fqn}.iso"
ISO_PATH="{automationWorkerBaseDirectory}/$ISO"
PROXMOX_USER="{proxmoxUser.user}"

if [[ ! -f "$ISO_PATH" ]]
then
    echo "Error: ISO file not found: $ISO_PATH"
    exit 1
fi

echo "Copying $ISO to Proxmox via SCP..."

scp -o StrictHostKeyChecking=no "$ISO_PATH" $PROXMOX_USER@$PVE_HOST:/var/lib/vz/template/iso/

if [[ $? -eq 0 ]]
then
    echo "ISO copied successfully: local:iso/$ISO"
else
    echo "Copy failed"
    exit 1
fi
